089            /**
090             * Creates a new database file, initializing the appropriate headers. Enough space is allocated in
091             * </br>the index for the specified initial size.
092             * @param dbPath - Pathname where the file is located as a String
093             * @param initialSize - Size of the db created
094             * @throws IOException
095             * @throws RecordsFileException
096             */
097            protected BaseRecordsFile(String dbPath, int initialSize) throws IOException, RecordsFileException {
098                    DATA_START_HEADER_LOCATION = 4;
099                    FILE_HEADERS_REGION_LENGTH = 16;
100                    RECORD_HEADER_LENGTH = 16;
101                    MAX_KEY_LENGTH = 64;
102                    INDEX_ENTRY_LENGTH = MAX_KEY_LENGTH + RECORD_HEADER_LENGTH;
103                    File f = sanityCheck(dbPath, true, true);
104                    file = new RandomAccessFile(f, "rw");
105                    dataStartPtr = indexPositionToEntryFp(initialSize);     // Record Data Region starts were the
106                    setFileLength(dataStartPtr);                                            // (i+1)th index entry would start.
107                    writeNumRecordsHeader(0);
108                    writeDataStartPtrHeader(dataStartPtr);
109            }

182            /**
183             * Checks if a file exists. If {@code create} is set to true and the file does not exist</br>
184             * it will try to create all necessary dirs and the file.
185             * @param dbPath - The path to the file
186             * @param create - If {@code true} creates the file plus dirs if it does not exist
187             * @param errorMsg - throws a RecordsFileException if the file already exists
188             * @throws RecordsFileException
189             * @throws IOException
190             * @return The file if successfully created
191             */
192            private File sanityCheck(String dbPath, boolean create, boolean errorMsg) throws RecordsFileException, IOException {
193                    if (dbPath == null)
194                            throw new RecordsFileException("Path can not be null");
195                    File f = new File(dbPath);
196                    if (create) {
197                            if (f.exists() && errorMsg)
198                                    throw new RecordsFileException("Database already exists: " + dbPath);
199                            if (createFile(f))
200                                    return f;
201                            else
202                                    throw new RecordsFileException("Failed to create database file at: " + dbPath);
203                    } else if (!create) {
204                            if (!f.exists() && errorMsg)
205                                    throw new RecordsFileException("Database not found: " + dbPath);
206                            return f;
207                    }
208                    return f;
209            }

211            /**
212             * Creates a file and if necessary all parent directories.
213             * @param dbFile - The file which needs to be created
214             * @throws IOException
215             * @return {@code true} if file could be created, otherwise {@code false}
216             */
217            private boolean createFile(File dbFile) throws IOException {
218                    dbFile.mkdirs();
219                    return dbFile.createNewFile();

311             * Returns a file pointer in the index pointing to the first byte
312             * in the key located at the given index position.

314            protected long indexPositionToEntryFp(int pos) {
315                    return FILE_HEADERS_REGION_LENGTH + (INDEX_ENTRY_LENGTH * pos);

319             * Reads the ith key from the index.

323                    file.seek(indexPositionToEntryFp(position));
324                    return file.readUTF();
325            }

328             * Reads the ith index entry from the index.

332                    file.seek(position);

336            /**
337             * Writes the ith index entry to the index.
338             * @throws IOException
339             */
340            protected void writeEntryToIndex(IndexEntry entry) throws IOException {
341                    file.seek(entry.indexPosition);
342                    entry.write(file);
343            }
344    
345            /**
346             * Appends an entry to end of index. Assumes that insureIndexSpace() has already been called.
347             * @throws RecordsFileException
348             * @throws IOException
349             */
350            protected void addEntryToIndex(IndexEntry newEntry, int currentNumRecords) throws IOException, RecordsFileException {
351                    file.seek(indexPositionToEntryFp(currentNumRecords));
352                    newEntry.write(file);
353                    newEntry.setIndexPosition(currentNumRecords);
354                    writeNumRecordsHeader(currentNumRecords + 1);
355            }
356    
357    
358            /**
359             * Removes the record from the index. Replaces the target with the entry at the
360             * end of the index.
361             * @throws RecordsFileException
362             * @throws IOException
363             */
364            protected void deleteEntryFromIndex(String key, IndexEntry entry, int currentNumRecords) throws IOException, RecordsFileException {
365                    if (entry.indexPosition != currentNumRecords - 1) {
366                            String lastKey = readKeyFromIndex(currentNumRecords - 1);
367                            IndexEntry last = keyToIndexEntry(lastKey);
368                            last.setIndexPosition(entry.indexPosition);
369                            file.seek(indexPositionToEntryFp(last.indexPosition));
370                            last.write(file);
371                    }
372                    writeNumRecordsHeader(currentNumRecords - 1);
373            }

375            /**
376             * Adds the given record to the database.
377             * @throws RecordsFileException
378             * @throws IOException
379             */
380            public synchronized void insertRecord(RecordWriter rw) throws RecordsFileException, IOException {
381                    String key = rw.getKey();
382                    if (recordExists(key)) {
383                            throw new RecordsFileException("Key exists: " + key);
384                    }
385                    insureIndexSpace(getNumRecords() + 1);
386                    IndexEntry newEntry = allocateRecord(key, rw.getDataLength());
387                    writeRecordData(newEntry, rw);
388                    addEntryToIndex(newEntry, getNumRecords());
389            }
390    
391            /**
392             * Appends the Record to the end of a file without searching for free space.
393             * @throws RecordsFileException
394             * @throws IOException
395             */
396            public synchronized void quickInsertRecord(RecordWriter rw) throws RecordsFileException, IOException {
397                    String key = rw.getKey();
398                    int dataLength = rw.getDataLength();
399                    insureIndexSpace(getNumRecords() + 1);
400                    long fp = getFileLength();
401                    setFileLength(fp + dataLength);
402                    IndexEntry newEntry = new IndexEntry(fp, dataLength, key, MAX_KEY_LENGTH);
403                    writeRecordData(newEntry, rw);
404                    addEntryToIndex(newEntry, getNumRecords());
405            }

407            /**
408             * Updates an existing record. If the new contents does not fit in the original record,
409             * then the update is handled by deleting the old record and adding the new.
410             * @throws RecordsFileException
411             * @throws IOException
412             */
413            public synchronized void updateRecord(RecordWriter rw) throws RecordsFileException, IOException {
414                    IndexEntry entry = keyToIndexEntry(rw.getKey());
415                    if (rw.getDataLength() > entry.dataCapacity) {
416                            deleteRecord(rw.getKey());
417                            insertRecord(rw);
418                    } else {
419                            writeRecordData(entry, rw);
420                            writeEntryToIndex(entry);
421                    }
422            }

443            /**
444             * Reads the record data for the given record header.
445             * @throws IOException
446             */
447            protected byte[] readRecordData(IndexEntry entry) throws IOException {
448                    byte[] buf = new byte[entry.dataCount];
449                    file.seek(entry.dataPointer);
450                    file.readFully(buf);
451                    return buf;
452            }

454            /**
455             * Updates the contents of the given record. A RecordsFileException is thrown if the new data does not
456             * fit in the space allocated to the record. The header's data count is updated, but not
457             * written to the file.
458             * @throws RecordsFileException
459             * @throws IOException
460             */
461            protected void writeRecordData(IndexEntry entry, RecordWriter rw) throws IOException, RecordsFileException {
462                    if (rw.getDataLength() > entry.dataCapacity) {
463                            throw new RecordsFileException ("Record data does not fit");
464                    }
465                    entry.dataCount = rw.getDataLength();
466                    file.seek(entry.dataPointer);
467                    rw.writeTo((DataOutput)file);
468            }

471            /**
472             * Updates the contents of the given record. A RecordsFileException is thrown if the new data does not
473             * fit in the space allocated to the record. The header's data count is updated, but not
474             * written to the file.
475             * @throws RecordsFileException
476             * @throws IOException
477             */
478            protected void writeRecordData(IndexEntry entry, byte[] data) throws IOException, RecordsFileException {
479                    if (data.length > entry.dataCapacity) {
480                            throw new RecordsFileException ("Record data does not fit");
481                    }
482                    entry.dataCount = data.length;
483                    file.seek(entry.dataPointer);
484                    file.write(data, 0, data.length);
485            }

488            /**
489             * Deletes a record.
490             * @throws RecordsFileException
491             * @throws IOException
492             */
493            public synchronized void deleteRecord(String key) throws RecordsFileException, IOException {
494                    IndexEntry delEntry = keyToIndexEntry(key);
495                    int currentNumRecords = getNumRecords();
496                    if (getFileLength() == delEntry.dataPointer + delEntry.dataCapacity) {
497                            // shrink file since this is the last record in the file
498                            setFileLength(delEntry.dataPointer);
499                    } else {
500                            IndexEntry previous = getIndexAt(delEntry.dataPointer - 1);
501                            if (previous != null) {
502                                    // append space of deleted record onto previous record
503                                    previous.dataCapacity += delEntry.dataCapacity;
504                                    writeEntryToIndex(previous);
505                            } else {
506                                    // target record is first in the file and is deleted by adding its space to
507                                    // the second record.
508                                    IndexEntry secondEntry = getIndexAt(delEntry.dataPointer + (long)delEntry.dataCapacity);
509                                    byte[] data = readRecordData(secondEntry);
510                                    secondEntry.dataPointer = delEntry.dataPointer;
511                                    secondEntry.dataCapacity += delEntry.dataCapacity;
512                                    writeRecordData(secondEntry, data);
513                                    writeEntryToIndex(secondEntry);
514                            }
515                    }
516                    deleteEntryFromIndex(key, delEntry, currentNumRecords);
517            }

520            /**
521             * Checks to see if there is space for and additional index entry. If
522             * not, space is created by moving records to the end of the file.
523             * @throws RecordsFileException
524             * @throws IOException
525             */
526            protected void insureIndexSpace(int requiredNumRecords) throws RecordsFileException, IOException {
527                    int originalFirstDataCapacity;
528                    int currentNumRecords = getNumRecords();
529                    long endIndexPtr = indexPositionToEntryFp(requiredNumRecords);
530                    if (endIndexPtr > getFileLength() && currentNumRecords == 0) {
531                            setFileLength(endIndexPtr);
532                            dataStartPtr = endIndexPtr;
533                            writeDataStartPtrHeader(dataStartPtr);
534                            return;
535                    }
536                    while (endIndexPtr > dataStartPtr) {
537                            IndexEntry first = getIndexAt(dataStartPtr);
538                            byte[] data = readRecordData(first);
539                            first.dataPointer = getFileLength();
540                            // If first.dataCapacity is set to the actual data count BEFORE resetting dataStartPtr,
541                            // and there is free space in 'first', then dataStartPtr will not be reset to the start
542                            // of the second record. Capture the capacity first and use it to perform the reset.
543                            originalFirstDataCapacity = first.dataCapacity;
544                            first.dataCapacity = data.length;
545                            setFileLength(first.dataPointer + data.length);
546                            writeRecordData(first, data);
547                            writeEntryToIndex(first);
548                            dataStartPtr += originalFirstDataCapacity;
549                            writeDataStartPtrHeader(dataStartPtr);
550                    }
551            }

553            /**
554             * Closes the file.
555             * @throws RecordsFileException
556             * @throws IOException
557             */
558            public synchronized void close() throws IOException, RecordsFileException {
559                    try {
560                            file.close();
561                    } finally {
562                            file = null;
563                    }
