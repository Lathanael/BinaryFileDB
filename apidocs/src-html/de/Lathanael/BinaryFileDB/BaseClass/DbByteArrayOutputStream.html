019    package de.Lathanael.BinaryFileDB.BaseClass;
020    
021    import java.io.*;
022    
023    /**
024     * Extends ByteArrayOutputStream to provide a way of writing the buffer to
025     * a DataOutput without re-allocating it.
026     * @author original by: Derek Hamner (http://www.javaworld.com/javaworld/jw-01-1999/jw-01-step.html?page=1)
027     * @author modified by: Lathanael (aka Philippe Leipold)
028     */
029    public class DbByteArrayOutputStream extends ByteArrayOutputStream {
030    
031            public DbByteArrayOutputStream() {
032                    super();
033            }
034    
035            public DbByteArrayOutputStream(int size) {
036                    super(size);
037            }
038    
039            /**
040             * Writes the full contents of the buffer to a DataOutput stream.
041             */
042            public synchronized void writeTo (DataOutput dstr) throws IOException {
043                    byte[] data = super.buf;
044                    int l = super.size();
045                    dstr.write(data, 0, l);
046            }
047    }
