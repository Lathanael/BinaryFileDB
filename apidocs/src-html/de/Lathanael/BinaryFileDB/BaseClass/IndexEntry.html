019    package de.Lathanael.BinaryFileDB.BaseClass;
020    
021    import java.io.*;
022    import java.nio.ByteBuffer;
023    
024    import de.Lathanael.BinaryFileDB.Exception.RecordsFileException;
025    
026    /**
027     * IndexEntry object for the DB-file index header
028     * @author Lathanael (aka Philippe Leipold)
029     */
030    public class IndexEntry {
031    
032            private JavaModUTFConverter converter = new JavaModUTFConverter();
033            /**
034             * File pointer to the first byte of record data (8 bytes).
035             */
036            protected long dataPointer;
037    
038            /**
039             * Actual number of bytes of data held in this record (4 bytes).
040             */
041            protected int dataCount;
042    
043            /**
044             * Number of bytes of data that this record can hold (4 bytes).
045             */
046            protected int dataCapacity;
047    
048            /**
049             * Indicates this entry's position in the file index.
050             */
051            protected int indexPosition;
052    
053            /**
054             * The key used to refer to this index entry
055             */
056            protected String key;
057    
058            /**
059             * The maximum length allowed for a key
060             */
061            protected int maxKeyLength;
062    
063            /**
064             * Empty constructor
065             */
066            protected IndexEntry() {
067            }
068    
069            /**
070             * Creates a new IndexEntry object
071             * @param dataPointer - Where in the file is the data located.
072             * @param dataCapacity - The maximum capacity for this entry
073             * @param key - The key with which this entry can be read
074             * @param maxKeyLength - The maximum KeyLength allowed by the {@link de.Lathanael.BinaryFileDB.BaseClass.RecordsFile RecordsFile}
075             */
076            protected IndexEntry(long dataPointer, int dataCapacity, String key, int maxKeyLength) {
077                    if (dataCapacity < 1)
078                            throw new IllegalArgumentException("Bad record size: " + dataCapacity);
079                    this.dataPointer = dataPointer;
080                    this.dataCapacity = dataCapacity;
081                    this.key = key;
082                    this.maxKeyLength = maxKeyLength;
083                    this.dataCount = 0;
084            }
085    
086            /**
087             * Gets the key of the entry as a String
088             */
089            protected String getKey() {
090                    return key;
091            }
092    
093            /**
094             * Set the Key of the entry. Be careful with this as it can corrupt the database-file.
095             */
096            protected void setKey(String key) {
097                    this.key = key;
098            }
099    
100            /**
101             * Get the position of this entry in the index.
102             */
103            protected int getIndexPosition() {
104                    return indexPosition;
105            }
106    
107            /**
108             * Set the index-position of the entry. Be careful with this as it can corrupt the database-file.
109             */
110            protected void setIndexPosition(int indexPosition) {
111                    this.indexPosition = indexPosition;
112            }
113    
114            /**
115             * Gets the maximum capacity associeated with this entry.
116             */
117            protected int getDataCapacity() {
118                    return dataCapacity;
119            }
120    
121            /**
122             * Gets the free space for this entry.
123             */
124            protected int getFreeSpace() {
125                    return dataCapacity - dataCount;
126            }
127    
128            protected void read(DataInput in) throws IOException {
129                    ByteBuffer buffer = ByteBuffer.allocate(16 + maxKeyLength);
130                    in.readFully(buffer.array(), 0, 16 + maxKeyLength);
131                    key = converter.readUTF(buffer);
132                    buffer.position(maxKeyLength);
133                    dataPointer = buffer.getLong();
134                    dataCapacity = buffer.getInt();
135                    dataCount = buffer.getInt();
136            }
137    
138            protected void write(DataOutput out) throws IOException {
139                    ByteBuffer buffer = ByteBuffer.allocate(16 + maxKeyLength);
140                    buffer.put(converter.writeUTF(key));
141                    buffer.position(maxKeyLength);
142                    buffer.putLong(dataPointer).putInt(dataCapacity).putInt(dataCount);
143                    out.write(buffer.array());
144            }
145    
146            protected static IndexEntry readEntry(DataInput in) throws IOException {
147                    IndexEntry entry = new IndexEntry();
148                    entry.read(in);
149                    return entry;
150            }
151    
152            /**
153             * Returns a new index entry which occupies the free space of this entry.
154             * Shrinks this entries size by the size of its free space.
155             */
156            protected IndexEntry split(String key) throws RecordsFileException {
157                    long newFp = dataPointer + (long)dataCount;
158                    IndexEntry newRecord = new IndexEntry(newFp, getFreeSpace(), key, maxKeyLength);
159                    dataCapacity = dataCount;
160                    return newRecord;
161            }
162    }
