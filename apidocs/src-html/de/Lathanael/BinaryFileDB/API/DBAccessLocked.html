037     * the use of a {@link de.Lathanael.BinaryFileDB.BaseClass.RecordsFile RecordsFile}.</br>
038     * It also provides some methods to handle multiple files of the</br>
039     * same type and a WriteQueue to lesser the I/O done by the</br>

059             * amount of space which is allocated for the index. The index can grow</br>
060             * dynamically, but the parameter is provided to increase efficiency.</br>
061             * The queue function is disabled.
062             * @param dbPath - Pathname where the file is located as a String
063             * @param initialSize - Size of the db created
064             * @throws CacheSizeException
065             * @throws IOException

072                    this.initialSize = initialSize;
073                    file = new RecordsFile(dbPath, initialSize, cacheSize);
074                    setMetrics();
075            }

632             * Gets a record as a {@link de.Lathanael.BinaryFileDB.API.RecordReader RecordReader} object from the DB.
633             * </p>
634             * If the WriteQueue is activated it will first check if the entry is scheduled and get the changed Record.
635             * @param key

637             * @throws RecordsFileException
638             * @throws IOException
639             * @throws InterruptedException
640             */
641            public RecordReader getRecord(String key) throws RecordsFileException, IOException, InterruptedException {
642                    reads++;
643                    RecordWriter rw;
644                    if (useQueue &&  (rw = queue.getQueuedItem(key)) != null)
645                            return new RecordReader(key, rw);
646                    lock.getReadLock();
647                    try {
648                            return file.readRecord(key);
649                    } finally {
650                            lock.releaseReadLock();
651                    }

654            /**
655             * Writes a Record to the DB. Existing ones will be updated.
656             *  </p>
657             * If the queue is activated it will try to add the RecordWriter to it.</br>
658             * If it fails it will write the whole queue to the DB, clear it and finally</br>
659             * add the given RecordWriter to the now empty queue.
660             * @param rw - RecordWriter object which should be written
661             * @throws RecordsFileException
662             * @throws IOException
663             * @throws QueueException
664             * @throws InterruptedException
665             */
666            public void writeRecord(RecordWriter rw) throws RecordsFileException, IOException, QueueException, InterruptedException {
667                    writes++;
668                    lock.getWriteLock();
669                    try {
670                            if (useQueue) {
671                                    if(!queue.addToQueue(rw)) {
672                                            for (Map.Entry<String, RecordWriter> entry: queue.getQueue().entrySet()) {
673                                                    boolean append = entry.getValue().isAppend();
674                                                    if (file.recordExists(entry.getValue().getKey())) {
675                                                            file.updateRecord(entry.getValue());
676                                                            continue;
677                                                    }
678                                                    if (append) {
679                                                            file.quickInsertRecord(entry.getValue());
680                                                            continue;
681                                                    }
682                                                    file.insertRecord(entry.getValue());
683                                            }
684                                            queue.clearQueue();
685                                            if (!queue.addToQueue(rw))
686                                                    throw new QueueException("Could not add Item to the queue");
687                                    }
688                                    file.removeFromCache(rw.getKey());
689                                    updateTimeStamp();
690                            } else {
691                                    if (file.recordExists(rw.getKey()))
692                                            file.updateRecord(rw);
693                                    else if (rw.isAppend())
694                                            file.quickInsertRecord(rw);
695                                    else
696                                            file.insertRecord(rw);
697                                    file.removeFromCache(rw.getKey());
698                                    updateTimeStamp();
699                            }
700                    } finally {
701                            lock.releaseWriteLock();
702                    }
703            }

705            /**
706             * Writes a Record to the DB without searching for free-space. Existing ones will be updated.
707             * </p>
708             * If the queue is activated it will try to add the RecordWriter to it.</br>
709             * If it fails it will write the whole queue to the DB, clear it and finally</br>
710             * add the given RecordWriter to the now empty queue.
711             * @param rw - RecordWriter object which should be written
712             * @throws IOException
713             * @throws RecordsFileException
714             * @throws QueueException
715             * @throws InterruptedException
716             */
717            public void appendRecord(RecordWriter rw) throws RecordsFileException, IOException, QueueException, InterruptedException {
718                    writes++;
719                    lock.getWriteLock();
720                    try {
721                            rw.setAppend(true);
722                            if (useQueue) {
723                                    if(!queue.addToQueue(rw)) {
724                                            for (Map.Entry<String, RecordWriter> entry: queue.getQueue().entrySet()) {
725                                                    if (file.recordExists(entry.getValue().getKey())) {
726                                                            file.updateRecord(entry.getValue());
727                                                            continue;
728                                                    }
729                                                    file.quickInsertRecord(entry.getValue());
730                                            }
731                                            queue.clearQueue();
732                                            if (!queue.addToQueue(rw))
733                                                    throw new QueueException("Could not add Item to the queue");
734                                    }
735                                    file.removeFromCache(rw.getKey());
736                                    updateTimeStamp();
737                            } else {
738                                    if (file.recordExists(rw.getKey()))
739                                            file.updateRecord(rw);
740                                    else
741                                            file.insertRecord(rw);
742                                    file.removeFromCache(rw.getKey());
743                                    updateTimeStamp();
744                            }
745                    } finally {
746                            lock.releaseWriteLock();
747                    }
748            }

850            /**
851             * Close the database safley. This will save all things in the queue before closing the file.
852             * @throws RecordsFileException
853             * @throws IOException
854             * @throws InterruptedException
855             */
856            private void safelyCloseDB() throws IOException, RecordsFileException, InterruptedException {
857                    lock.getWriteLock();
858                    try {
859                            if (useQueue) {
860                                    for (Map.Entry<String, RecordWriter> entry: queue.getQueue().entrySet()) {
861                                            writes++;
862                                            if (file.recordExists(entry.getValue().getKey())) {
863                                                    file.updateRecord(entry.getValue());
864                                                    continue;
865                                            }
866                                            file.insertRecord(entry.getValue());
867                                    }
868                            }
869                    } finally {
870                            lock.releaseWriteLock();
871                            file.close();
872                    }
873            }
