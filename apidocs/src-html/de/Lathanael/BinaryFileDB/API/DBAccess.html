033     * This class provides an API to a BinaryFile-DataBase through</br>
034     * the use of a {@link de.Lathanael.BinaryFileDB.BaseClass.RecordsFile RecordsFile}.</br>
035     * It also provides some methods to handle multiple files of the</br>
036     * same type and a WriteQueue to lesser the I/O done by the</br>
037     * file-database.

049            /**
050             * Creates a new database file. The initialSize parameter determines the</br>
051             * amount of space which is allocated for the index. The index can grow</br>
052             * dynamically, but the parameter is provided to increase efficiency.</br>
053             * The queue function is disabled.
054             * @param dbPath - Pathname where the file is located as a String
055             * @param initialSize - Size of the db created
056             * @throws CacheSizeException
057             * @throws IOException
058             * @throws RecordsFileException
059             */
060            public DBAccess (String dbPath, int initialSize) throws IOException, RecordsFileException, CacheSizeException {
061                    useQueue = false;
062                    cacheSize = 10;
063                    this.initialSize = initialSize;
064                    file = new RecordsFile(dbPath, initialSize, cacheSize);
065            }

518            /**
519             * Writes a Record to the DB. Existing ones will be updated.
520             *  </p>
521             * If the queue is activated it will try to add the RecordWriter to it.</br>
522             * If it fails it will write the whole queue to the DB, clear it and finally</br>
523             * add the given RecordWriter to the now empty queue.
524             * @param rw - RecordWriter object which should be written
525             * @throws RecordsFileException
526             * @throws IOException
527             * @throws QueueException
528             */
529            public void writeRecord(RecordWriter rw) throws RecordsFileException, IOException, QueueException {
530                    if (useQueue) {
531                            if(!queue.addToQueue(rw)) {
532                                    for (Map.Entry<String, RecordWriter> entry: queue.getQueue().entrySet()) {
533                                            boolean append = entry.getValue().isAppend();
534                                            if (file.recordExists(entry.getValue().getKey())) {
535                                                    file.updateRecord(entry.getValue());
536                                                    continue;
537                                            }
538                                            if (append) {
539                                                    file.quickInsertRecord(entry.getValue());
540                                                    continue;
541                                            }
542                                            file.insertRecord(entry.getValue());
543                                    }
544                                    queue.clearQueue();
545                                    if (!queue.addToQueue(rw))
546                                            throw new QueueException("Could not add Item to the queue");
547                            }
548                    } else {
549                            if (file.recordExists(rw.getKey()))
550                                    file.updateRecord(rw);
551                            else if (rw.isAppend())
552                                    file.quickInsertRecord(rw);
553                            else
554                                    file.insertRecord(rw);
555                    }
556                    file.removeFromCache(rw.getKey());
557            }

559            /**
560             * Writes a Record to the DB without searching for free-space. Existing ones will be updated.
561             * </p>
562             * If the queue is activated it will try to add the RecordWriter to it.</br>
563             * If it fails it will write the whole queue to the DB, clear it and finally</br>
564             * add the given RecordWriter to the now empty queue.
565             * @param rw - RecordWriter object which should be written
566             * @throws IOException
567             * @throws RecordsFileException
568             * @throws QueueException
569             */
570            public void appendRecord(RecordWriter rw) throws RecordsFileException, IOException, QueueException {
571                    rw.setAppend(true);
572                    if (useQueue) {
573                            if(!queue.addToQueue(rw)) {
574                                    for (Map.Entry<String, RecordWriter> entry: queue.getQueue().entrySet()) {
575                                            if (file.recordExists(entry.getValue().getKey())) {
576                                                    file.updateRecord(entry.getValue());
577                                                    continue;
578                                            }
579                                            file.quickInsertRecord(entry.getValue());
580                                    }
581                                    queue.clearQueue();
582                                    if (!queue.addToQueue(rw))
583                                            throw new QueueException("Could not add Item to the queue");
584                            }
585                    } else {
586                            file.quickInsertRecord(rw);
587                    }
588                    file.removeFromCache(rw.getKey());
589            }

647            /**
648             * Close the database safley. This will save all things in the queue before closing the file.
649             * @throws RecordsFileException
650             * @throws IOException
651             */
652            private void safelyCloseDB() throws IOException, RecordsFileException {
653                    if (useQueue) {
654                            for (Map.Entry<String, RecordWriter> entry: queue.getQueue().entrySet()) {
655                                    if (file.recordExists(entry.getValue().getKey())) {
656                                            file.updateRecord(entry.getValue());
657                                            continue;
658                                    }
659                                    file.insertRecord(entry.getValue());
660                            }
661                    }
662                    file.close();
663            }

667             * @throws IOException
668             * @throws RecordsFileException

671                    file.close();
672            }
