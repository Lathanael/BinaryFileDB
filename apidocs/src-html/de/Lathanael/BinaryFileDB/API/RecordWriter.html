019    package de.Lathanael.BinaryFileDB.API;
020    
021    import java.io.*;
022    
023    import de.Lathanael.BinaryFileDB.BaseClass.DbByteArrayOutputStream;
024    
025    /**
026     * A class to serialize an object first and call the write-process later.
027     * @author original by: Derek Hamner (http://www.javaworld.com/javaworld/jw-01-1999/jw-01-step.html?page=1)
028     * @author modified by: Lathanael (aka Philippe Leipold)
029     */
030    public class RecordWriter {
031    
032            private String key;
033            private DbByteArrayOutputStream out;
034            private ObjectOutputStream objOutStr;
035            private Object objOut;
036            private boolean append = false;
037    
038            public RecordWriter(String key) {
039                    this.key = key;
040                    out = new DbByteArrayOutputStream();
041            }
042    
043            public RecordWriter(String key, boolean append) {
044                    this.key = key;
045                    out = new DbByteArrayOutputStream();
046                    this.append = append;
047            }
048    
049            public String getKey() {
050                    return key;
051            }
052    
053            public OutputStream getOutputStream() {
054                    return out;
055            }
056    
057            public Object getObjectOut() {
058                    return objOut;
059            }
060    
061            public void setAppend(boolean append) {
062                    this.append = append;
063            }
064    
065            public boolean isAppend() {
066                    return append;
067            }
068    
069            public ObjectOutputStream getObjectOutputStream() throws IOException {
070                    if (objOutStr == null) {
071                            objOutStr = new ObjectOutputStream(out);
072                    }
073                    return objOutStr;
074            }
075    
076            /**
077             * Serializes an object so it can be written to a file.
078             * @param o They object which should be serialized
079             * @throws IOException
080             */
081            public void writeObject(Object o) throws IOException {
082                    objOut = o;
083                    getObjectOutputStream().writeObject(o);
084                    getObjectOutputStream().flush();
085            }
086    
087            /**
088             * Returns the number of bytes in the data.
089             */
090            public int getDataLength() {
091                    return out.size();
092            }
093    
094            /**
095             * Writes the data out to the stream without re-allocating the buffer.
096             */
097            public void writeTo(DataOutput str) throws IOException {
098                    out.writeTo(str);
099            }
100    }
