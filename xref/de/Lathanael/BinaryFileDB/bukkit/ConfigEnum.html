21  package de.Lathanael.BinaryFileDB.bukkit;
22  
23  import java.util.LinkedHashMap;
24  import java.util.List;
25  import java.util.Map;
26  
27  import org.bukkit.configuration.Configuration;
28  import org.bukkit.plugin.PluginDescriptionFile;
29  
30  /***
31   * @author Lathanael (aka Philippe Leipold)
32   *
33   */
34  public enum ConfigEnum {
35  
36  	DEBUG("debug", true, "if true, a log-file will be created in the folder of the plugin");
37  	public final static String PREFIX = "ac_";
38  	private final String confVal;
39  	private final Object defaultVal;
40  	private final String description;
41  	private static Configuration config;
42  	private static String pluginVersion;
43  	private static String pluginName;
44  
45  	/***
46  	 * @param confVal
47  	 * @param defaultVal
48  	 * @param description
49  	 */
50  	private ConfigEnum(final String confVal, final Object defaultVal, final String description) {
51  		this.confVal = confVal;
52  		this.defaultVal = defaultVal;
53  		this.description = description;
54  	}
55  
56  	public String getString() {
57  		return config.getString(confVal);
58  	}
59  
60  	public int getInt() {
61  		return config.getInt(confVal);
62  	}
63  
64  	public double getDouble() {
65  		return config.getDouble(confVal);
66  	}
67  
68  	public boolean getBoolean() {
69  		return config.getBoolean(confVal);
70  	}
71  
72  	public long getLong() {
73  		return config.getLong(confVal);
74  	}
75  
76  	public float getFloat() {
77  		return Float.parseFloat(config.getString(confVal));
78  	}
79  
80  	public List<String> getStringList() {
81  		return config.getStringList(confVal);
82  	}
83  
84  	public void setValue(final Object value) {
85  		config.set(confVal, value);
86  	}
87  
88  	/***
89  	 * @return the defaultvalues
90  	 */
91  	public static Map<String, Object> getDefaultvalues() {
92  		final Map<String, Object> values = new LinkedHashMap<String, Object>();
93  		for (final ConfigEnum ce : values())
94  			values.put(ce.confVal, ce.defaultVal);
95  		return values;
96  	}
97  
98  	public static String getHeader() {
99  		final StringBuffer buffer = new StringBuffer();
100 		buffer.append("Configuration file of ").append(pluginName).append('\n');
101 		buffer.append("Plugin Version: ").append(pluginVersion).append('\n').append('\n');
102 		for (final ConfigEnum ce : values())
103 			buffer.append(ce.confVal).append("\t:\t").append(ce.description).append(" (Default : ")
104 					.append(ce.defaultVal).append(')').append('\n');
105 		return buffer.toString();
106 	}
107 
108 	/***
109 	 * @param config
110 	 *            the config to set
111 	 */
112 	public static void setConfig(final Configuration config) {
113 		ConfigEnum.config = config;
114 	}
115 
116 	public static void setPluginInfos(final PluginDescriptionFile pdf) {
117 		ConfigEnum.pluginVersion = pdf.getVersion();
118 		ConfigEnum.pluginName = pdf.getName();
119 	}
120 }
