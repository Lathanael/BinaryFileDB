19  package de.Lathanael.BinaryFileDB.BaseClass;
20  
21  import java.io.*;
22  import java.nio.ByteBuffer;
23  
24  import de.Lathanael.BinaryFileDB.Exception.RecordsFileException;
25  
26  /***
27   * IndexEntry object for the DB-file index header
28   * @author Lathanael (aka Philippe Leipold)
29   */
30  public class IndexEntry {
31  
32  	private JavaModUTFConverter converter = new JavaModUTFConverter();
33  	/***
34  	 * File pointer to the first byte of record data (8 bytes).
35  	 */
36  	protected long dataPointer;
37  
38  	/***
39  	 * Actual number of bytes of data held in this record (4 bytes).
40  	 */
41  	protected int dataCount;
42  
43  	/***
44  	 * Number of bytes of data that this record can hold (4 bytes).
45  	 */
46  	protected int dataCapacity;
47  
48  	/***
49  	 * Indicates this entry's position in the file index.
50  	 */
51  	protected int indexPosition;
52  
53  	/***
54  	 * The key used to refer to this index entry
55  	 */
56  	protected String key;
57  
58  	/***
59  	 * The maximum length allowed for a key
60  	 */
61  	protected int maxKeyLength;
62  
63  	/***
64  	 * Empty constructor
65  	 */
66  	protected IndexEntry() {
67  	}
68  
69  	/***
70  	 * Creates a new IndexEntry object
71  	 * @param dataPointer - Where in the file is the data located.
72  	 * @param dataCapacity - The maximum capacity for this entry
73  	 * @param key - The key with which this entry can be read
74  	 * @param maxKeyLength - The maximum KeyLength allowed by the {@link de.Lathanael.BinaryFileDB.BaseClass.RecordsFile RecordsFile}
75  	 */
76  	protected IndexEntry(long dataPointer, int dataCapacity, String key, int maxKeyLength) {
77  		if (dataCapacity < 1)
78  			throw new IllegalArgumentException("Bad record size: " + dataCapacity);
79  		this.dataPointer = dataPointer;
80  		this.dataCapacity = dataCapacity;
81  		this.key = key;
82  		this.maxKeyLength = maxKeyLength;
83  		this.dataCount = 0;
84  	}
85  
86  	/***
87  	 * Gets the key of the entry as a String
88  	 */
89  	protected String getKey() {
90  		return key;
91  	}
92  
93  	/***
94  	 * Set the Key of the entry. Be careful with this as it can corrupt the database-file.
95  	 */
96  	protected void setKey(String key) {
97  		this.key = key;
98  	}
99  
100 	/***
101 	 * Get the position of this entry in the index.
102 	 */
103 	protected int getIndexPosition() {
104 		return indexPosition;
105 	}
106 
107 	/***
108 	 * Set the index-position of the entry. Be careful with this as it can corrupt the database-file.
109 	 */
110 	protected void setIndexPosition(int indexPosition) {
111 		this.indexPosition = indexPosition;
112 	}
113 
114 	/***
115 	 * Gets the maximum capacity associeated with this entry.
116 	 */
117 	protected int getDataCapacity() {
118 		return dataCapacity;
119 	}
120 
121 	/***
122 	 * Gets the free space for this entry.
123 	 */
124 	protected int getFreeSpace() {
125 		return dataCapacity - dataCount;
126 	}
127 
128 	protected void read(DataInput in) throws IOException {
129 		ByteBuffer buffer = ByteBuffer.allocate(16 + maxKeyLength);
130 		in.readFully(buffer.array(), 0, 16 + maxKeyLength);
131 		key = converter.readUTF(buffer);
132 		buffer.position(maxKeyLength);
133 		dataPointer = buffer.getLong();
134 		dataCapacity = buffer.getInt();
135 		dataCount = buffer.getInt();
136 	}
137 
138 	protected void write(DataOutput out) throws IOException {
139 		ByteBuffer buffer = ByteBuffer.allocate(16 + maxKeyLength);
140 		buffer.put(converter.writeUTF(key));
141 		buffer.position(maxKeyLength);
142 		buffer.putLong(dataPointer).putInt(dataCapacity).putInt(dataCount);
143 		out.write(buffer.array());
144 	}
145 
146 	protected static IndexEntry readEntry(DataInput in) throws IOException {
147 		IndexEntry entry = new IndexEntry();
148 		entry.read(in);
149 		return entry;
150 	}
151 
152 	/***
153 	 * Returns a new index entry which occupies the free space of this entry.
154 	 * Shrinks this entries size by the size of its free space.
155 	 */
156 	protected IndexEntry split(String key) throws RecordsFileException {
157 		long newFp = dataPointer + (long)dataCount;
158 		IndexEntry newRecord = new IndexEntry(newFp, getFreeSpace(), key, maxKeyLength);
159 		dataCapacity = dataCount;
160 		return newRecord;
161 	}
162 }
