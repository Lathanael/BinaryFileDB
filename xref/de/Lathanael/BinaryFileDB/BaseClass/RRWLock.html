23  import java.util.HashMap;
24  import java.util.Map;

28  /***
29   * An implementation of a fully reentrant Read-Lock and Write-Lock
30   * @author original by: Jakob Jenkov (http://java.dzone.com/news/java-concurrency-read-write-lo?page=0,1)
31   * @author modified by: Lathanael (aka Philippe Leipold)
32   */
33  public class RRWLock {
34  
35  	private Map<Thread, Counter> readingThreads = new HashMap<Thread, Counter>();
36  
37  	private int writeAccesses = 0;
38  	private int writeRequests  = 0;
39  	private Thread writingThread = null;
40  
41  
42  	/***
43  	 * Get a read Lock
44  	 * @throws InterruptedException
45  	 */
46  	public synchronized void getReadLock() throws InterruptedException {
47  		Thread callingThread = Thread.currentThread();
48  		while(!canRead(callingThread))
49  			wait();
50  
51  		Counter c = getReaderCount(callingThread);
52  		c.count++;
53  		readingThreads.put(callingThread, c);
54  	}
55  
56  	/***
57  	 * Release a read Lock
58  	 */
59  	public synchronized void releaseReadLock() {
60  		Thread callingThread = Thread.currentThread();
61  		if(!isReader(callingThread))
62  			throw new IllegalMonitorStateException("Calling Thread does not hold a read lock on this ReadWriteLock");
63  		Counter c = getReaderCount(callingThread);
64  		if(c.count == 1)
65  			readingThreads.remove(callingThread);
66  		else
67  			c.count--;
68  
69  		notifyAll();
70  	}
71  
72  	/***
73  	 * Get a write Lock
74  	 * @throws InterruptedException
75  	 */
76  	public synchronized void getWriteLock() throws InterruptedException {
77  		writeRequests++;
78  		Thread callingThread = Thread.currentThread();
79  		while(!canWrite(callingThread))
80  			wait();
81  
82  		writeRequests--;
83  		writeAccesses++;
84  		writingThread = callingThread;
85  	}
86  
87  	/***
88  	 * Release a write Lock
89  	 * @throws InterruptedException
90  	 */
91  	public synchronized void releaseWriteLock() throws InterruptedException {
92  		if(!isWriter(Thread.currentThread()))
93  			throw new IllegalMonitorStateException("Calling Thread does not hold the write lock on this ReadWriteLock");
94  
95  		writeAccesses--;
96  		if(writeAccesses == 0)
97  			writingThread = null;
98  
99  		notifyAll();
100 	}
101 
102 	private boolean canRead(Thread callingThread) {
103 		if(isWriter(callingThread))
104 			return true;
105 		if(hasWriter())
106 			return false;
107 		if(isReader(callingThread))
108 			return true;
109 		if(hasRequests())
110 			return false;
111 		return true;
112 	}
113 
114 	private boolean canWrite(Thread callingThread) {
115 		if(isReaderOnly(callingThread))
116 			return true;
117 		if(hasReaders())
118 			return false;
119 		if(writingThread == null)
120 			return true;
121 		if(!isWriter(callingThread))
122 			return false;
123 		return true;
124 	}
125 
126 	private Counter getReaderCount(Thread callingThread) {
127 		Counter accessCount = readingThreads.get(callingThread);
128 		if(accessCount == null)
129 			return new Counter();
130 
131 		return accessCount;
132 	}
133 
134 	private boolean isWriter(Thread callingThread) {
135 		return writingThread == callingThread;
136 	}
137 
138 	private boolean isReader(Thread callingThread) {
139 		return readingThreads.get(callingThread) != null;
140 	}
141 
142 	private boolean isReaderOnly(Thread callingThread) {
143 		return readingThreads.size() == 1 && readingThreads.get(callingThread) != null;
144 	}
145 
146 	private boolean hasWriter() {
147 		return writingThread != null;
148 	}
149 
150 	private boolean hasReaders() {
151 		return readingThreads.size() > 0;
152 	}
153 
154 	private boolean hasRequests() {
155 		return writeRequests > 0;
156 	}
157 }
