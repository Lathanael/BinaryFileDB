19  package de.Lathanael.BinaryFileDB.API;
20  
21  import java.io.*;
22  
23  /***
24   * Deserializes an object from an InputStream
25   * @author original by: Derek Hamner (http://www.javaworld.com/javaworld/jw-01-1999/jw-01-step.html?page=1)
26   * @author modified by: Lathanael (aka Philippe Leipold)
27   */
28  public class RecordReader {
29  
30  	private String key;
31  	private byte[] data;
32  	private ByteArrayInputStream in;
33  	private ObjectInputStream objIn;
34  	private Object obj;
35  
36  	public RecordReader(String key, byte[] data) {
37  		this.key = key;
38  		this.data = data;
39  		in = new ByteArrayInputStream(data);
40  	}
41  
42  	public RecordReader(String key, RecordWriter rw) {
43  		this.key = key;
44  		obj = rw.getObjectOut();
45  	}
46  
47  	public String getKey() {
48  		return key;
49  	}
50  
51  	public byte[] getData() {
52  		return data;
53  	}
54  
55  	public Object getObjectIn() {
56  		return obj;
57  	}
58  
59  	public InputStream getInputStream() throws IOException {
60  		return in;
61  	}
62  
63  	public ObjectInputStream getObjectInputStream() throws IOException {
64  		if (objIn == null)
65  			objIn = new ObjectInputStream(in);
66  		return objIn;
67  	}
68  
69  	/***
70  	 * Reads the next object in the record using an ObjectInputStream.
71  	 */
72  	public Object readObject() throws IOException, OptionalDataException, ClassNotFoundException {
73  		return getObjectInputStream().readObject();
74  	}
75  }
