35  /***
36   * This class provides an API to a BinaryFile-DataBase through</br>
37   * the use of a {@link de.Lathanael.BinaryFileDB.BaseClass.RecordsFile RecordsFile}.</br>
38   * It also provides some methods to handle multiple files of the</br>
39   * same type and a WriteQueue to lesser the I/O done by the</br>
40   * file-database and a ReentrantReadWriteLock.
41   * @author Lathanael (aka Philippe Leipold)
42   */
43  public class DBAccessLocked {

46  	private long TIMESTAMP;

51  	private int initialSize = 16;
52  	private int MAX_KEY_LENGTH = 64, DATA_START_HEADER_LOCATION = 4, RECORD_HEADER_LENGTH = 16, FILE_HEADERS_REGION_LENGTH = 16;

57  	/***
58  	 * Creates a new database file. The initialSize parameter determines the</br>
59  	 * amount of space which is allocated for the index. The index can grow</br>
60  	 * dynamically, but the parameter is provided to increase efficiency.</br>
61  	 * The queue function is disabled.
62  	 * @param dbPath - Pathname where the file is located as a String
63  	 * @param initialSize - Size of the db created
64  	 * @throws CacheSizeException
65  	 * @throws IOException
66  	 * @throws RecordsFileException
67  	 */
68  	public DBAccessLocked (String dbPath, int initialSize) throws IOException, RecordsFileException, CacheSizeException {
69  		TIMESTAMP = System.currentTimeMillis();
70  		useQueue = false;
71  		cacheSize = 10;
72  		this.initialSize = initialSize;
73  		file = new RecordsFile(dbPath, initialSize, cacheSize);
74  		setMetrics();
75  	}

654 	/***
655 	 * Writes a Record to the DB. Existing ones will be updated.
656 	 *  </p>
657 	 * If the queue is activated it will try to add the RecordWriter to it.</br>
658 	 * If it fails it will write the whole queue to the DB, clear it and finally</br>
659 	 * add the given RecordWriter to the now empty queue.
660 	 * @param rw - RecordWriter object which should be written
661 	 * @throws RecordsFileException
662 	 * @throws IOException
663 	 * @throws QueueException
664 	 * @throws InterruptedException
665 	 */
666 	public void writeRecord(RecordWriter rw) throws RecordsFileException, IOException, QueueException, InterruptedException {
667 		writes++;
668 		lock.getWriteLock();
669 		try {
670 			if (useQueue) {
671 				if(!queue.addToQueue(rw)) {
672 					for (Map.Entry<String, RecordWriter> entry: queue.getQueue().entrySet()) {
673 						boolean append = entry.getValue().isAppend();
674 						if (file.recordExists(entry.getValue().getKey())) {
675 							file.updateRecord(entry.getValue());
676 							continue;
677 						}
678 						if (append) {
679 							file.quickInsertRecord(entry.getValue());
680 							continue;
681 						}
682 						file.insertRecord(entry.getValue());
683 					}
684 					queue.clearQueue();
685 					if (!queue.addToQueue(rw))
686 						throw new QueueException("Could not add Item to the queue");
687 				}
688 				file.removeFromCache(rw.getKey());
689 				updateTimeStamp();
690 			} else {
691 				if (file.recordExists(rw.getKey()))
692 					file.updateRecord(rw);
693 				else if (rw.isAppend())
694 					file.quickInsertRecord(rw);
695 				else
696 					file.insertRecord(rw);
697 				file.removeFromCache(rw.getKey());
698 				updateTimeStamp();
699 			}
700 		} finally {
701 			lock.releaseWriteLock();
702 		}
703 	}
