33  /***
34   * This class provides an API to a BinaryFile-DataBase through</br>
35   * the use of a {@link de.Lathanael.BinaryFileDB.BaseClass.RecordsFile RecordsFile}.</br>
36   * It also provides some methods to handle multiple files of the</br>
37   * same type and a WriteQueue to lesser the I/O done by the</br>
38   * file-database and a ReentrantReadWriteLock.
39   * @author Lathanael (aka Philippe Leipold)
40   */
41  public class DBAccessLocked {
42  
43  	private final RecordsFile file;
44  	private boolean useQueue;
45  	private RRWLock lock = new RRWLock();
46  	private DataWriteQueue queue;
47  	private final int cacheSize;
48  	private int initialSize = 16;
49  	private int MAX_KEY_LENGTH = 64, DATA_START_HEADER_LOCATION = 4, RECORD_HEADER_LENGTH = 16, FILE_HEADERS_REGION_LENGTH = 16;

51  	/***
52  	 * Creates a new database file. The initialSize parameter determines the</br>
53  	 * amount of space which is allocated for the index. The index can grow</br>
54  	 * dynamically, but the parameter is provided to increase efficiency.</br>
55  	 * The queue function is disabled.
56  	 * @param dbPath - Pathname where the file is located as a String
57  	 * @param initialSize - Size of the db created
58  	 * @throws CacheSizeException
59  	 * @throws IOException
60  	 * @throws RecordsFileException
61  	 */
62  	public DBAccessLocked (String dbPath, int initialSize) throws IOException, RecordsFileException, CacheSizeException {
63  		useQueue = false;
64  		cacheSize = 10;
65  		this.initialSize = initialSize;
66  		file = new RecordsFile(dbPath, initialSize, cacheSize);
67  	}

533 	/***
534 	 * Writes a Record to the DB. Existing ones will be updated.
535 	 *  </p>
536 	 * If the queue is activated it will try to add the RecordWriter to it.</br>
537 	 * If it fails it will write the whole queue to the DB, clear it and finally</br>
538 	 * add the given RecordWriter to the now empty queue.
539 	 * @param rw - RecordWriter object which should be written
540 	 * @throws RecordsFileException
541 	 * @throws IOException
542 	 * @throws QueueException
543 	 * @throws InterruptedException
544 	 */
545 	public void writeRecord(RecordWriter rw) throws RecordsFileException, IOException, QueueException, InterruptedException {
546 		lock.getWriteLock();
547 		try {
548 			if (useQueue) {
549 				if(!queue.addToQueue(rw)) {
550 					for (Map.Entry<String, RecordWriter> entry: queue.getQueue().entrySet()) {
551 						boolean append = entry.getValue().isAppend();
552 						if (file.recordExists(entry.getValue().getKey())) {
553 							file.updateRecord(entry.getValue());
554 							continue;
555 						}
556 						if (append) {
557 							file.quickInsertRecord(entry.getValue());
558 							continue;
559 						}
560 						file.insertRecord(entry.getValue());
561 					}
562 					queue.clearQueue();
563 					if (!queue.addToQueue(rw))
564 						throw new QueueException("Could not add Item to the queue");
565 				}
566 				file.removeFromCache(rw.getKey());
567 			} else {
568 				if (file.recordExists(rw.getKey()))
569 					file.updateRecord(rw);
570 				else if (rw.isAppend())
571 					file.quickInsertRecord(rw);
572 				else
573 					file.insertRecord(rw);
574 				file.removeFromCache(rw.getKey());
575 			}
576 		} finally {
577 			lock.releaseWriteLock();
578 		}
579 	}
